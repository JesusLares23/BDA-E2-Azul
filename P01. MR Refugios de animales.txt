Modelo Relacional: Sistema de Registro de Voluntariado en Refugios de Animales




Voluntarios {
        id_voluntario : entero PK,
        nombre : texto(100),
        edad : entero,
        especialidad : texto(20),
        correo : texto(50),
        telefono : texto(20)
}


Refugios {
        id_refugio : entero PK,
        nombre : texto(50),
        ubicación : point,
        capacidad : entero,
        nombre_responsable : texto(100)
}


Animales {
        id_animal : entero PK,
        nombre : texto(30),
        especie : texto(30),
        edad : entero,
        estado_salud : texto(20),
        fecha_ingreso : fecha y hora,
        id_refugio : entero FK
                que pertenece a Refugios
}


Tarea {
        id_tarea : entero PK,
        fecha : fecha y hora,
        actividad : texto (50),
        id_voluntario : entero FK
                que pertenece a Voluntarios,
        id_animal : entero FK 
que pertenece a Animales,
id_refugio : entero FK
        que pertenece a Refugios
}
